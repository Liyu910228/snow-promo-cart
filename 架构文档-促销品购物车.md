## 三、数据流向

### 3.1 加入购物车

```
用户操作 → WarehouseSelect 组件
                ↓
        调用 useCartStore.addItem()
                ↓
        保存到 localStorage（持久化）
                ↓
        Pinia 响应式更新 → UI 自动刷新
```

### 3.2 拼单进度计算

```
后端返回拼单数据 → useWarehouseStore
                          ↓
                计算进度 = (currentQty + 本次数量) / minOrderQty
                          ↓
                判断状态 = currentQty >= minOrderQty ? 'ready' : 'pending'
                          ↓
                ProgressBar 组件渲染
```

### 3.3 推荐算法

```
获取所有仓储的拼单数据
          ↓
计算推荐分数 = (minOrderQty - currentQty) / minOrderQty
          ↓（越小越推荐，已拼越多越推荐）
排序，取分数最低的作为推荐仓
          ↓
WarehouseSelect 渲染时，推荐仓置顶
```

### 6.1 路由守卫

```typescript
// 路由守卫逻辑
router.beforeEach((to, from, next) => {
  const authStore = useAuthStore()
  
  // 未登录跳转登录页
  if (!authStore.isLoggedIn && to.path !== '/') {
    return next('/')
  }
  
  // 角色权限校验
  if (to.meta.role && authStore.userRole !== to.meta.role) {
    // 管理员进入用户页面 → 跳转到管理后台
    if (authStore.userRole === 'admin') {
      return next('/admin')
    }
    // 用户进入管理页面 → 跳转到首页
    return next('/')
  }
  
  next()
})
```

## 七、关键实现点

### 7.1 拼单进度计算

```typescript
// 仓储级进度（单仓拼单）
function calcWarehouseProgress(currentQty: number, minDeliveryQty: number, thisQty: number) {
  const total = currentQty + thisQty;
  const percent = Math.min((total / minDeliveryQty) * 100, 100);
  const status = total >= minDeliveryQty ? 'ready' : 'pending';
  return { total, percent, status, remaining: Math.max(minDeliveryQty - total, 0) };
}

// 全局进度（跨仓汇总）
function calcGlobalProgress(warehouseTotals: number[], minOrderQty: number) {
  const total = warehouseTotals.reduce((sum, qty) => sum + qty, 0);
  const percent = Math.min((total / minOrderQty) * 100, 100);
  const status = total >= minOrderQty ? 'ready' : 'pending';
  return { total, percent, status, remaining: Math.max(minOrderQty - total, 0) };
}
```

### 7.2 推荐仓计算

```typescript
// 仓储进度 = 已拼量 / 该仓最小起送量
// 推荐逻辑：距该仓最小起送量差距最小的优先
function getRecommendedWarehouse(warehouses: Warehouse[], groupOrders: GroupOrder[]) {
  const scored = warehouses.map(w => {
    const order = groupOrders.find(o => o.warehouseId === w.id);
    const currentQty = order?.currentQty || 0;
    const minDeliveryQty = order?.minDeliveryQty || 30;  // 该仓的最小起送量
    const remaining = minDeliveryQty - currentQty;       // 还差多少满足该仓
    return { ...w, score: remaining, currentQty, minDeliveryQty, orderId: order?.id };
  });
  // 按差值升序排列（已成团的 score=-1，优先）
  return scored.sort((a, b) => a.score - b.score);
}
```

### 7.3 数量校验

```typescript
// 用户下单校验
function validateQuantity(quantity: number) {
  if (!quantity || quantity < 1) {
    return { valid: false, message: '请输入下单数量' };
  }
  return { valid: true };
}
// 注意：最小起送量、最小起订量由系统内部判断，用户下单无感知限制
```
